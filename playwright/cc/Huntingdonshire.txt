⏺ Here's a minimal Playwright script example:

  import { chromium } from 'playwright';

  async function getBinDates(postcode, address) {
    const browser = await chromium.launch();
    const page = await browser.newPage();

    // Navigate to calendar
    await page.goto('https://www.huntingdonshire.gov.uk/refuse-calendar/');

    // Close popup
    await page.click('button:has-text("Save and Close")');

    // Enter postcode and search
    await page.fill('input[placeholder*="postcode"]', postcode);
    await page.click('button:has-text("Search")');

    // Select address
    await page.selectOption('select', address);

    // View calendar
    await page.click('a:has-text("View Calendar")');

    // Extract dates
    const waste = await page.textContent(':has-text("domestic waste") strong');
    const recycling = await page.textContent(':has-text("dry recycling") strong');

    console.log('Waste:', waste);
    console.log('Recycling:', recycling);

    await browser.close();
  }

  getBinDates('PE19 0AA', '69 NUFFIELD ROAD, ST NEOTS, PE190AA');

  That's the core flow in ~25 lines. Real production code would add error handling, waits, and better selectors.
